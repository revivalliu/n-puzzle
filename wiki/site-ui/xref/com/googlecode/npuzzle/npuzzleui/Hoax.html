20  public class Hoax extends StackPane {

22      private int width = 110;
23      private int height = 110;
24      private int value = 0;
25  
26      public Hoax(int width, int height, int value) {
27          this.width = width;
28          this.height = height;
29          this.value = value;
30  
31          drawHoax();
32      }
33  
34      public Hoax() {
35          drawHoax();
36      }
37  
38      private void drawHoax() {
39          setSize(new Dimensions(width, height));
40          Drawing drawing = new Drawing();
41          Canvas canvas = new Canvas();
42          Rectangle rectangle = new Rectangle();
43          rectangle.setHeight(height);
44          rectangle.setWidth(width);
45          rectangle.setFill("{paintType:'gradient', startX:0, startY:0, startColor:'#009900', endX:0, endY:240, endColor:'#000000'}");
46          rectangle.setCornerRadius(10);
47          rectangle.setStroke("#FFFFFF");
48          rectangle.setStrokeThickness(1);
49          canvas.add(rectangle);
50          drawing.setCanvas(canvas);
51          ImageView imageView = new ImageView(drawing);
52          imageView.setStyles("{horizontalAlignment:'left', verticalAlignment:'top'}");
53          Label label = new Label();
54          label.setText(" " + value + " ");
55          label.setStyles("{color:'#FFFFFF', horizontalAlignment:'center', verticalAlignment:'center'}");
56          label.getStyles().put("font", new Font("Arial", Font.PLAIN, (int) height / 3));
57          add(imageView);
58          add(label);
59      }
