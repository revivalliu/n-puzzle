7   import org.apache.pivot.wtk.FlowPane;
8   import org.apache.pivot.wtk.Label;
9   import org.apache.pivot.wtk.TablePane;
10  
11  /**
12   *
13   * @author thiago
14   */
15  public class StatusBar extends TablePane {
16  
17      private Label label = new Label("Stoped");
18      private Label commandBuffer = new Label("Command buffer: ");
19  
20      public static enum Status {
21  
22          SOLVING, MOVING, STOPED
23      };
24      private Status status;
25  
26      public StatusBar(int width) {
27          getColumns().add(new Column(width / 2));
28          getColumns().add(new Column(width / 2));
29          FlowPane flowA = new FlowPane();
30          FlowPane flowB = new FlowPane();
31          flowA.add(label);
32          flowB.add(commandBuffer);
33          Row row = new Row();
34          row.add(flowA);
35          row.add(flowB);
36          getRows().add(row);
37      }
38  
39      public void setBufferSize(int size) {
40          commandBuffer.setText("Command buffer: " + size);
41      }
42  
43      public void setStatus(Status desc) {
44          status = desc;
45          String text = "Stoped";
46          switch (desc) {
47              case MOVING:
48                  text = "Moving";
49                  break;
50              case SOLVING:
51                  text = "Solving...";
52                  break;
53              case STOPED:
54                  text = "Stoped";
55                  break;
56          }
57          label.setText(text);
58      }
59  
60      public Status getStatus() {
61          return status;
62      }
63  }
